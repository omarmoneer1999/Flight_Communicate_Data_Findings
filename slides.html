tot_delays = df_8['WeatherDelay'].sum() + df_8['CarrierDelay'].sum()+df_8['SecurityDelay'].sum()+df_8['NASDelay'].sum()+df_8['LateAircraftDelay'].sum()
#We will work on the calculation 
wd = (df_8['WeatherDelay'].sum())/tot_delays
cd = (df_8['CarrierDelay'].sum())/tot_delays
sd = (df_8['SecurityDelay'].sum())/tot_delays
nd = (df_8['NASDelay'].sum())/tot_delays
ld = (df_8['LateAircraftDelay'].sum())/tot_delays
delay_ratios = [wd,cd,sd,nd,ld]

#Here we will work on making the visualization and the titles. 
lab = ['Weather Delays','Carrier Delays' , 'NAS Delays', 'Secuirity Delays','Late Air Craft']
plt.pie(delay_ratios,labels = lab, autopct='%.2f%%');
